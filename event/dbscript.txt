CREATE PROPERTY Blog.parent STRING;
CREATE PROPERTY Blog.children STRING;
CREATE PROPERTY Blog.attributes STRING;
CREATE PROPERTY Blog.upUsers STRING;
CREATE PROPERTY Blog.downUsers STRING;
CREATE PROPERTY Blog.rank STRING;
CREATE PROPERTY Blog.posts STRING;
CREATE PROPERTY Blog.createDate STRING;
CREATE PROPERTY Blog.createUserId STRING;
